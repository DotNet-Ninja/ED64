   7:
   8:Unit vic;
   9:(****************************************************************)
  10:interface
  11:
  12:uses SystemAbstraction;
  13:
  14:procedure vicText(pos: word; val: byte);
  15:function  vicRead(address : word) : byte;
  16:procedure vicWrite(address : word ; val : byte);
  17:procedure vicBank(val : byte);
  18:(****************************************************************)
  19:
  20:implementation
  21:
  22:uses memio;
  23:
  24:
  25:var
  26:    bankAddress : word;
  27:    screenAddress : word;
  28:    charRomAddress: word;
  29:    reg : array[$D000..$D02E] of byte;
  30:
  31:procedure vicBank(val : byte);
  32:begin
  33:    bankAddress := not(val and 3) * $4000;
  34:    setVicwatchRange(bankAddress+screenAddress,bankAddress+screenAddress+999);   
  35:end;
  36:
  37:function  vicRead(address : word) : byte;
  38:begin
  39:    case address of
  40:        $D012 : vicRead := $00; //raster
  41:    else
  42:        vicRead := reg[address]
  43:    end
  44:end;
  45:
  46:procedure vicWrite(address : word ; val : byte);
  47:begin
  48:    reg[address] := val;
  49:    case address of
  50:        $D018 : begin
  51:                    screenAddress := (val shr 4)*$400;
  52:                    setVicwatchRange(bankAddress+screenAddress,bankAddress+screenAddress+999);
  53:                    charRomAddress := (val and $0E)*$400;
  54:                end
  55:    end
  56:end;
  57:
  58:procedure vicText(pos: word; val: byte);
  59:var x0,y0,x,y:integer;
  60:begin
  61:       x0:=(pos mod 40)*8;
  62:       y0:=(pos div 40)*8;
  63:       for x:=0 to 7 do
  64:          for y := 0 to 7 do
  65:             if boolean(vicPeek(bankAddress+charRomAddress+val*8+y) and ($80 shr x))
  66:                   then setPixel(x0+x,y0+y,14) //lightblue
  67:                   else setPixel(x0+x,y0+y,6)  //blue
  68:
  69:
  70:end;
  71:
  72:
  73:
  74:end.
