  28:uses WinGraph, WinCRT, memio;

  30:var gd,gm : integer;
  31:    PixelSize : byte;
  32:
  33:procedure openscreen;
  34:var res : char;
  35:begin
  36:    writeln ('Choose graphic resolution');
  37:    writeln;
  38:    writeln ('0. Default       4. 1024x768');
  39:    writeln ('1. 320x200       5. 1280x1024');
  40:    writeln ('2. 640x480       6. Maximized');
  41:    writeln ('3. 800x600       7. Fullscreen');
  42:    writeln;
  43:    writeln ('Your choise : (another value sets to Default) ');
  44:    readln (res);
  45:    case res of
  46:        '1' : gm := m320x200;
  47:        '2' : gm := m640x480;
  48:        '3' : gm := m800x600;
  49:        '4' : gm := m1024x768;
  50:        '5' : gm := m1280x1024;
  51:        '6' : gm := mMaximized;
  52:        '7' : gm := mFullscr
  53:        else
  54:            gm := mDefault
  55:    end;
  56:
  57:    gd := D8bit;
  58:
  59:    initgraph(gd,gm,'ED64');
  60:        if graphresult <> grOK then
  61:        begin
  62:                        writeln('graphic mode not supported');
  63:                        readln;
  64:                        halt(1);
  65:    end;
  66:    pixelsize := (getMaxX+1) div 320;
  67:
  68:    SetBkColor(14);
  69:    ClearDevice;
  70:    setviewport((getmaxX-320*pixelSize) div 2,(getMaxY-200*pixelSize) div 2,getmaxX,getMaxY,clipoff);
  71:
  72:    setPalette(0,black);
  73:    setPalette(1,white);
  74:    setPalette(2,red);
  75:    setPalette(3,cyan);
  76:    setPalette(4,purple);
  77:    setPalette(5,green);
  78:    setPalette(6,blue);
  79:    setPalette(7,yellow);
  80:    setPalette(8,orange);
  81:    setPalette(9,brown);
  82:    setPalette(10,lightred);
  83:    setPalette(11,darkgray);
  84:    setPalette(12,gray);
  85:    setPalette(13,lightgreen);
  86:    setPalette(14,lightblue);
  87:    setPalette(15,lightgray)
  88:end;
  89:
  90:procedure closeScreen;
  91:begin
  92:    closeGraph
  93:end;

 100:end;
