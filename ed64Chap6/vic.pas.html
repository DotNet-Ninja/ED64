  17:uses memio,SystemAbstraction;
  18:
  19:
  20:
  21:procedure vicText(pos: word; val: byte);
  22:var x0,y0,x,y:integer;
  23:begin
  24:       x0:=(pos mod 40)*8;
  25:       y0:=(pos div 40)*8;
  26:       for x:=0 to 7 do
  27:          for y := 0 to 7 do
  28:             if boolean(vicPeek($1000+val*8+y) and ($80 shr x))
  29:                   then setPixel(x0+x,y0+y,14) //lightblue
  30:                   else setPixel(x0+x,y0+y,6)  //blue
  31:
  32:end;
  33:
  34:
  35:
  36:end.
