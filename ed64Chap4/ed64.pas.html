  57:(******************************************************************
  58:* load and save routines                                          *

  60:procedure load_prg (filename : string ; basic : boolean);
  61:var     f : file of byte;
  62:        b : byte;
  63:        address : word;
  64:begin
  65:        {$i-}
  66:        assign (f,filename);
  67:        reset(f);
  68:        {$i+}
  69:        if ioresult <> 0
  70:        then
  71:                writeln('No file named ',filename)
  72:        else
  73:           begin
  74:                read(f,b);
  75:                address := b;
  76:                read(f,b);
  77:                address := address + b*256;
  78:                if basic then address := 2048;
  79:                while not eof(f) do
  80:                    begin
  81:                        read(f,b);
  82:                        poke(address,b);
  83:                        inc(address)
  84:                    end;
  85:                close(f);
  86:                poke ($2D,lo(address));
  87:                poke ($2E,hi(address));
  88:                PC := $A52A
  89:           end;
  90:end;
  91:
  92:procedure save_prg (filename : string ; from_address,to_address : word);
  93:var     f : file of byte;
  94:        i : word;
  95:begin
  96:        assign (f,filename);
  97:        rewrite(f);
  98:        write(f,lo(from_address));
  99:        write(f,hi(from_address));
 100:        for i := from_address to to_address do write(f,peek(i));
 101:        close(f)
 102:end;

 104:(******************************************************************

 107:procedure addkey;
 108:var     buff,ch : byte;
 109:        filename: string;
 110:begin
 111:        buff := peek($C6);
 112:        if buff < 10 then
 113:                begin   ch := ord(upcase(readkey));
 114:                        if ch = 0 then
 115:                           begin
 116:                                ch := ord(readkey);
 117:                                case ch of
 118:                                $48 : ch := 145; //up
 119:                                $4b : ch := 157; //left
 120:                                $50 : ch :=  17; //down
 121:                                $4D : ch :=  29; //right
 122:                                 82 : ch := 148; //inst-del
 123:                                133 : begin
 124:                                        writeln;
 125:                                        write('EMULATOR LOAD. FILENAME: ');
 126:                                        readln (filename);
 127:                                        load_prg (filename+'.prg',true)
 128:                                      end;
 129:                                135 : begin
 130:                                        writeln;
 131:                                        write('EMULATOR LOAD,x,1. FILENAME: ');
 132:                                        readln (filename);
 133:                                        load_prg (filename+'.prg',false)
 134:                                      end;
 135:                                134 : begin
 136:                                        writeln;
 137:                                        write('EMULATOR SAVE. FILENAME: ');
 138:                                        readln (filename);
 139:                                        save_prg (filename+'.prg',2048,peek2($2D))
 140:                                      end;
 141:                                end;
 142:                           end;
 143:                        poke($0277+buff,ch);
 144:                        poke($C6,buff+1)
 145:                end
 146:
 147:end;

 226:procedure setflag (flag : byte; status : boolean);
 227:begin
 228:        if status       then P := P or flag
 229:                        else P := P and not flag
 230:end;

 232:function flagset (flag : byte) : boolean;
 233:begin
 234:        flagset := boolean(P and flag)
 235:end;

 237:(********************************************************************
 238:* Stack operations                                                  *
 239:********************************************************************)
 240:
 241:procedure push (b : byte);
 242:begin
 243:        poke ($100+S,b);
 244:        dec(S)
 245:end;
 246:
 247:function pull : byte;
 248:begin
 249:        inc(S);
 250:        pull := peek($100+S)
 251:end;

 253:(******************************************************************
 254:* generic branching instructions                                  *
 255:******************************************************************)
 256:procedure bfs (flag : byte);
 257:begin
 258:        if flagset(flag)   then PC := PC + shortint(peek(PC)) + 1
 259:                           else inc(PC)
 260:end;
 261:
 262:procedure bfc (flag : byte);
 263:begin
 264:        if flagset(flag)   then inc(PC)
 265:                           else PC := PC + shortint(peek(PC)) + 1
 266:end;

 272:procedure adc (address : word);
 273:var H : word;
 274:begin
 275:        H := A + peek(address);
 276:        if flagset(C) then H := H + 1;
 277:        A := H;
 278:        setflag(C,H > $FF);
 279:        setflag(Z,A = 0);
 280:        setflag(N,A >= $80);
 281:        H := H + $80;
 282:        setflag(V,(H > $FF) or (H < 0));
 283:end;
 284:
 285:procedure sbc (address : word);
 286:var H : word;
 287:begin
 288:        H := A - peek(address);
 289:        if not flagset(C) then H := H - 1;
 290:        A := H;
 291:        setflag(C,H<=$FF);
 292:        setflag(Z,A=0);
 293:        setflag(N,A>=$80);
 294:        H := H + $80;
 295:        setflag(V,(H > $FF) or (H < 0));
 296:end;
