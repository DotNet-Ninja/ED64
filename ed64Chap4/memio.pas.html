   7:unit memio;
   8:
   9:(****************************************************************)
  10:interface
  11:
  12:procedure poke(address : word ; value : byte);
  13:function  peek(address : word) : byte;
  14:(****************************************************************)
  15:
  16:
  17:implementation
  18:
  19:uses vic;
  20:
  21:var
  22:    ram         : array [$0000..$FFFF] of byte;
  23:    basic_rom   : array [$A000..$BFFF] of byte;
  24:    kernal_rom  : array [$E000..$FFFF] of byte;
  25:
  26:function io(address : word) : byte;
  27:begin
  28:        case address of
  29:                $D012 : io := $00
  30:        else
  31:                io := $FF
  32:        end
  33:end;
  34:
  35:procedure poke(address : word ; value : byte);
  36:begin
  37:    ram[address] := value;
  38:    case address of
  39:        $400..$7E7 : VicText(address-$400,value)
  40:    end
  41:end;
  42:
  43:function peek(address : word) : byte;
  44:begin
  45:    case address of
  46:        $A000..$BFFF : peek := basic_rom[address];
  47:        $D000..$DFFF : peek := io(address);
  48:        $E000..$FFFF : peek := kernal_rom[address]
  49:    else
  50:        peek := ram[address]
  51:    end
  52:end;
  53:
  54:
  55:
  56:procedure loadrom (var rom : array of byte ; filename: string);
  57:var
  58:    f : file of byte;
  59:    b : byte;
  60:    address: word;
  61:begin
  62:        {$i-}
  63:        assign (f,filename);
  64:        reset(f);
  65:        {$i+}
  66:        if ioresult <> 0 then
  67:                begin
  68:                        writeln('file ',filename,' is missing');
  69:                        readln;
  70:                        halt
  71:                end;
  72:        address := low(rom);
  73:        while not eof(f) do
  74:        begin
  75:                read(f,b);
  76:                rom[address] := b;
  77:                inc(address)
  78:        end;
  79:        close(f)
  80:end;
  81:
  82:initialization
  83:begin
  84:    loadrom(kernal_rom,'kernal.rom');
  85:    loadrom(basic_rom,'basic.rom')
  86:end;
  87:
  88:end.
