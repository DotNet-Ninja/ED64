   6:program ED64;
   7:
   8:uses memio;
   9:
  10:(*****************************************************************
  11:* 6510 registers                                                 *
  12:*****************************************************************)
  13:var
  14:        A,X,Y,S,P,IR : byte;
  15:        PC : word;
  16:
  17:(******************************************************************
  18:* Emulator help procedures and functions                                                  *
  19:******************************************************************)
  20:procedure error (s: string);
  21:begin
  22:    writeln('--------------------------------------');
  23:    writeln('emulator error: ',s);
  24:    writeln('PC=',hexstr(PC-1,4),' IR=',hexstr(IR,2));
  25:    writeln;
  26:    writeln('Execution has been ended');
  27:    writeln('push return to exit');
  28:    writeln('--------------------------------------');
  29:    readln;
  30:    halt
  31:end;
  32:
  33:(******************************************************************
  34:* processor main loop                                             *
  35:******************************************************************)
  36:begin
  37:    writeln('--------------------------------------');
  38:    writeln('Welcome to EL DENDO''s c64 emulator');
  39:    writeln('(c) 2006 ir. Marc Dendooven');
  40:    writeln('--------------------------------------');
  41:
  42:    poke($0000,$01);
  43:    poke($0001,$00);
  44:    poke($0002,$03);
  45:    PC := $0000;
  46:
  47:    while true do
  48:        begin
  49:                IR := peek(PC);
  50:                inc(PC);
  51:                case IR of
  52:                        $00 : writeln ('instruction 00');
  53:                        $01 : writeln ('instruction 01');
  54:                else
  55:                        error ('unknown instruction ')
  56:                end
  57:        end
  58:end.
