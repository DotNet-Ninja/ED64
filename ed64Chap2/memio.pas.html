   7:unit memio;

  10:interface
  11:
  12:procedure poke(address : word ; value : byte);
  13:function  peek(address : word) : byte;
  14:(****************************************************************)
  15:
  16:implementation
  17:
  18:var
  19:    ram : array [0..$FFFF] of byte;
  20:
  21:procedure poke(address : word ; value : byte);
  22:begin
  23:    ram[address] := value;
  24:end;
  25:
  26:function peek(address : word) : byte;
  27:begin
  28:    peek := ram[address]
  29:end;
  30:
  31:end.
