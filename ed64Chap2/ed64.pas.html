  24:procedure error (s: string);
  25:begin
  26:    writeln;
  27:    writeln('--------------------------------------');
  28:    writeln('emulator error: ',s);
  29:    writeln('PC=',hexstr(PC-1,4),' IR=',hexstr(IR,2));
  30:    writeln;
  31:    writeln('Execution has been ended');
  32:    writeln('push return to exit');
  33:    writeln('--------------------------------------');
  34:    readln;
  35:    halt
  36:end;

  38:function peek2(address : word) : word;
  39:begin
  40:        peek2 := peek(address) + peek(address + 1) * 256
  41:end;
  42:
  43:procedure dump1;
  44:begin
  45:        write('PC=',hexstr(PC,4),' IR=',hexstr(IR,2));
  46:end;
  47:
  48:procedure dump2;
  49:begin
  50:        writeln(' A=',hexstr(A,2),' X=',hexstr(X,2),' Y=',hexstr(Y,2),
  51:        ' S=',hexstr(S,2),' P=',binstr(P,8));
  52:end;

  54:(******************************************************************
  55:* Addressing modes: functions which return the address for        *
  56:* instructions to work upon                                       *
  57:******************************************************************)
  58:function imm : word;
  59:begin
  60:        imm := PC;
  61:        inc(PC)
  62:end;

  78:        zpy := peek(PC)+Y;
  79:        inc(PC)
  80:end;
  81:
  82:function abs : word;
  83:begin
  84:        abs := peek2(PC);
  85:        inc(PC,2)
  86:end;
  87:
  88:function absx : word;
  89:begin
  90:        absx := peek2(PC) + X;
  91:        inc(PC,2)
  92:end;
  93:

 100:function ind : word;
 101:begin
 102:        ind := peek2(peek2(PC));
 103:        inc(PC,2)
 104:end;
 105:
 106:function indx : word;
 107:begin
 108:        indx := peek2(byte(peek(PC)+X));
 109:        inc(PC)

 112:function indy : word;

 118:(********************************************************************
 119:* Actions on Status Register (P)    NV.BDIZC                        *
 120:********************************************************************)
 121:
 122:const
 123:        C = $01;
 124:        Z = $02;
 125:        I = $04;
 126:        D = $08;
 127:        B = $10;
 128:        V = $40;
 129:        N = $80;
 130:
 131:procedure setflag (flag : byte; status : boolean);
 132:begin
 133:        if status       then P := P or flag
 134:                        else P := P and not flag
 135:end;
 136:
 137:function flagset (flag : byte) : boolean;
 138:begin
 139:        flagset := boolean(P and flag)
 140:end;
 141:
 142:(********************************************************************
 143:* Stack operations                                                  *
 144:********************************************************************)
 145:
 146:procedure push (b : byte);
 147:begin
 148:        poke ($100+S,b);
 149:        dec(S)
 150:end;
 151:
 152:function pull : byte;
 153:begin
 154:        inc(S);
 155:        pull := peek($100+S)
 156:end;
 157:
 158:(******************************************************************
 159:* generic branching instructions                                  *
 160:******************************************************************)
 161:procedure bfs (flag : byte);
 162:begin
 163:        if flagset(flag)   then PC := PC + shortint(peek(PC)) + 1
 164:                           else inc(PC)
 165:end;
 166:
 167:procedure bfc (flag : byte);
 168:begin
 169:        if flagset(flag)   then inc(PC)
 170:                           else PC := PC + shortint(peek(PC)) + 1
 171:end;
 172:
 173:(********************************************************************
 174:* Instructions                                                      *
 175:********************************************************************)
 176:
 177:procedure adc (address : word);
 178:var H : word;
 179:begin
 180:        H := A + peek(address);
 181:        if flagset(C) then H := H + 1;
 182:        A := H;
 183:        setflag(C,H > $FF);
 184:        setflag(Z,A = 0);
 185:        setflag(N,A >= $80);
 186:        H := H + $80;
 187:        setflag(V,(H > $FF) or (H < 0));
 188:end;
 189:
 190:procedure sbc (address : word);
 191:var H : word;
 192:begin
 193:        H := A - peek(address);
 194:        if not flagset(C) then H := H - 1;
 195:        A := H;
 196:        setflag(C,H<=$FF);
 197:        setflag(Z,A=0);
 198:        setflag(N,A>=$80);
 199:        H := H + $80;
 200:        setflag(V,(H > $FF) or (H < 0));
 201:end;
 202:
 203:procedure and_ (address : word);
 204:begin
 205:        A := A and peek(address);
 206:        setflag(Z,A=0);
 207:        setflag(N,A>=$80);
 208:end;
 209:
 210:procedure ora (address : word);
 211:begin
 212:        A := A or peek(address);
 213:        setflag(Z,A=0);
 214:        setflag(N,A>=$80);
 215:end;
 216:
 217:procedure asl_A;
 218:begin
 219:        setflag(C,boolean(A and $80));
 220:        A := A shl 1;
 221:        setflag(Z,A=0);
 222:        setflag(N,A>=$80);
 223:end;
 224:
 225:procedure asl (address : word);
 226:var B : byte;
 227:begin
 228:        B := peek(address);
 229:        setflag(C,boolean(B and $80));
 230:        B := B shl 1;
 231:        poke(address,B);
 232:        setflag(Z,B=0);
 233:        setflag(N,B>=$80)
 234:end;
 235:
 236:procedure rol_A;
 237:var bit : boolean;
 238:begin
 239:        bit := flagset(C);
 240:        setflag(C,boolean(A and $80));
 241:        A := A shl 1;
 242:        if bit then A := A or $01;
 243:        setflag(Z,A=0);
 244:        setflag(N,A>=$80);
 245:end;
 246:
 247:procedure rol (address : word);
 248:var B : byte;
 249:    bit : boolean;
 250:begin
 251:        B := peek(address);
 252:        bit := flagset(C);
 253:        setflag(C,boolean(B and $80));
 254:        B := B shl 1;
 255:        if bit then B := B or $01;
 256:        poke(address,B);
 257:        setflag(Z,B=0);
 258:        setflag(N,B>=$80)
 259:end;
 260:
 261:procedure ror_A;
 262:var bit : boolean;
 263:begin
 264:        bit := flagset(C);
 265:        setflag(C,boolean(A and $01));
 266:        A := A shr 1;
 267:        if bit then A := A or $80;
 268:        setflag(Z,A=0);
 269:        setflag(N,A>=$80);
 270:end;
 271:
 272:procedure ror (address : word);
 273:var B : byte;
 274:    bit : boolean;
 275:begin
 276:        B := peek(address);
 277:        bit := flagset(C);
 278:        setflag(C,boolean(B and $01));
 279:        B := B shr 1;
 280:        if bit then B := B or $80;
 281:        poke(address,B);
 282:        setflag(Z,B=0);
 283:        setflag(N,B>=$80)
 284:end;
 285:
 286:procedure lsr_A;
 287:begin
 288:        setflag(C,boolean(A and $01));
 289:        A := A shr 1;
 290:        setflag(Z,A=0);
 291:        setflag(N,false);
 292:end;
 293:
 294:procedure lsr (address : word);
 295:var B : byte;
 296:begin
 297:        B := peek(address);
 298:        setflag(C,boolean(B and $01));
 299:        B := B shr 1;
 300:        poke(address,B);
 301:        setflag(Z,B=0);
 302:        setflag(N,false);
 303:end;
 304:
 305:procedure bit (address : word);
 306:var H : byte;
 307:begin
 308:        H := peek(address);
 309:        setflag(N,boolean(H and $80));
 310:        setflag(V,boolean(H and $40));
 311:        setflag(Z,(H and A)=0)
 312:end;
 313:
 314:
 315:
 316:procedure brk;
 317:begin
 318:        inc(PC);
 319:        push(hi(PC));
 320:        push(lo(PC));
 321:        dec(PC);
 322:        push(P);
 323:        setflag(I,true)
 324:end;
 325:
 326:procedure cmp (address : word);
 327:var H : word;
 328:begin
 329:        H := A - peek(address);
 330:        setflag(C,H<=$FF);
 331:        setflag(Z,lo(H)=0);
 332:        setflag(N,lo(H)>=$80)
 333:end;
 334:
 335:procedure cpx (address : word);
 336:var H : word;
 337:begin
 338:        H := X - peek(address);
 339:        setflag(C,H<=$FF);
 340:        setflag(Z,lo(H)=0);
 341:        setflag(N,lo(H)>=$80)
 342:end;
 343:
 344:procedure cpy (address : word);
 345:var H : word;
 346:begin
 347:        H := Y - peek(address);
 348:        setflag(C,H<=$FF);
 349:        setflag(Z,lo(H)=0);
 350:        setflag(N,lo(H)>=$80)
 351:end;
 352:
 353:procedure dec_ (address : word);
 354:var H : word;
 355:begin
 356:        H := peek(address) - 1;
 357:        poke (address,H);
 358:        setflag(Z,lo(H)=0);
 359:        setflag(N,lo(H)>=$80)
 360:end;
 361:
 362:procedure dex;
 363:var H : word;
 364:begin
 365:        H := X - 1;
 366:        X := H;
 367:        setflag(Z,X=0);
 368:        setflag(N,X>=$80)
 369:end;
 370:
 371:procedure dey;
 372:var H : word;
 373:begin
 374:        H := Y - 1;
 375:        Y := H;
 376:        setflag(Z,lo(H)=0);
 377:        setflag(N,lo(H)>=$80)
 378:end;
 379:
 380:procedure eor (address : word);
 381:begin
 382:        A := A xor peek(address);
 383:        setflag(Z,A=0);
 384:        setflag(N,A>=$80);
 385:end;

 388:var H : word;

 393:        setflag(N,lo(H)>=$80)

 402:        setflag(N,X>=$80)

 409:        Y := H;

 411:        setflag(N,Y>=$80)

 414:procedure jmp (address : word);
 415:begin
 416:        PC := address;
 417:end;
 418:
 419:procedure jsr (address : word);
 420:begin
 421:        dec(PC);
 422:        push(hi(PC));
 423:        push(lo(PC));
 424:        PC := address
 425:end;
 426:
 427:procedure rts;
 428:begin
 429:        PC := pull;
 430:        PC := PC + pull*256 + 1
 431:end;
 432:
 433:procedure rti;

 435:        P := pull;
 436:        PC := pull;
 437:        PC := PC + pull*256
 438:end;
